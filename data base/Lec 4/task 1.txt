    NAME VARCHAR2(50)
);

SELECT NAME, TRIM(NAME) AS TRIMMED_NAME FROM EMP_TEST;
------------------------------------------------------
SELECT NAME, LTRIM(NAME) AS NO_LEADING_SPACES FROM EMP_TEST;
-------------------------------------------------------------
SELECT NAME, RTRIM(NAME) AS NO_TRAILING_SPACES FROM EMP_TEST;
----------------------------------------------------------------
-- Remove '#' from both sides
SELECT NAME, TRIM(BOTH '#' FROM NAME) AS REMOVED_HASH FROM EMP_TEST;

-- Remove '*' from both sides
SELECT NAME, TRIM(BOTH '*' FROM NAME) AS REMOVED_STAR FROM EMP_TEST;
